#!/bin/sh
set -eu

# Operation validation
operation="${1:-}"
case "$operation" in
    get) route="fill" ;;
    store) route="approve" ;;
    erase) route="reject" ;;
    "") exit 0 ;;
    *) exit 0 ;;
esac

# Token resolution (Precedence: GIT_CRED_PROXY_TOKEN > GIT_CRED_PROXY_TOKEN_FILE > Default)
if [ -n "${GIT_CRED_PROXY_TOKEN:-}" ]; then
    token="$GIT_CRED_PROXY_TOKEN"
elif [ -n "${GIT_CRED_PROXY_TOKEN_FILE:-}" ]; then
    if [ ! -f "$GIT_CRED_PROXY_TOKEN_FILE" ]; then
        printf "Error: GIT_CRED_PROXY_TOKEN_FILE is set to '%s' but file does not exist\n" "$GIT_CRED_PROXY_TOKEN_FILE" >&2
        exit 1
    fi
    token=$(cat "$GIT_CRED_PROXY_TOKEN_FILE")
else
    default_token_file="/run/host-git-cred-proxy/token"
    if [ ! -f "$default_token_file" ]; then
        printf "Error: No token found. Checked GIT_CRED_PROXY_TOKEN, GIT_CRED_PROXY_TOKEN_FILE, and default %s\n" "$default_token_file" >&2
        exit 1
    fi
    token=$(cat "$default_token_file")
fi

# Trim whitespace/newlines from token
token=$(printf '%s' "$token" | tr -d ' \n\r')

# Proxy URL
proxy_url="${GIT_CRED_PROXY_URL:-http://host.docker.internal:18765}"

# HTTP request
# We use a temp file for the response body to correctly handle large outputs and separate stdout/stderr
resp_body=$(mktemp)
trap 'rm -f "$resp_body"' EXIT

# Perform POST request via curl
# -s: silent
# -S: show errors
# -X POST: use POST method
# -H: headers
# --data-binary @-: read body from stdin
# --write-out: capture HTTP status code
# -o: output response body to temp file
http_code=$(curl -s -S \
    -X POST \
    -H "Authorization: Bearer $token" \
    -H "Content-Type: text/plain; charset=utf-8" \
    --data-binary @- \
    --write-out "%{http_code}" \
    -o "$resp_body" \
    "$proxy_url/$route")

if [ "$http_code" = "200" ]; then
    if [ -s "$resp_body" ]; then
        cat "$resp_body"
    fi
    exit 0
else
    if [ -s "$resp_body" ]; then
        cat "$resp_body" >&2
    else
        printf "Proxy request failed with status %s\n" "$http_code" >&2
    fi
    exit 1
fi
